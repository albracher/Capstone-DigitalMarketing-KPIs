# clean up the delimiters in the audience column

dat_clean <- dat_clean |> 
  mutate(
    # 1. fixing the common encoding error by replacing the corrupted 'Â»' with a clean delimiter '>'
    audience_segment = str_replace_all(audience_segment, "Â»", ">"),
    
    # 2. replacing the remaining double-chevron ' »' (which sometimes has a hidden space) with a clean delimiter '>'
    audience_segment = str_replace_all(audience_segment, " »", ">"), 
    
    # 3. removing the unnecessary '<bb>' tag/separator entirely
    audience_segment = str_replace_all(audience_segment, "<bb>", ""),
    
    # 4. removing any extra spaces left behind after the replacements
    audience_segment = str_squish(audience_segment) 
  )

# normalize spacing around the cleaned delimiters to remove all adjacent whitespace

dat_clean <- dat_clean |> 
  mutate(
    # 1. remove all spaces adjacent to the delimiter. 
    #    this is the core step that makes ' a > b ' or 'a> b' become 'a>b', 
    #    resulting in the clean 'level1>level2>level3' format.
    audience_segment = stringr::str_replace_all(audience_segment, "\\s*\\>\\s*", ">"),
    
    # note: the step that added a single space back has been removed 
    #       to achieve the final "no whitespace" goal for easy separation.
    
    # 2. final cleanup: removing any leading or trailing spaces from the whole string
    audience_segment = stringr::str_trim(audience_segment),
    
    # 3. squishing any instances of multiple spaces left in the values themselves (e.g. 'united  states')
    audience_segment = stringr::str_squish(audience_segment) 
  )

# # start with the cleaned data
dat_clean_sep_state <- dat_clean |>
  # # step 1: initial split of location into city_raw and state_raw
  separate(
    col = location,
    into = c("city_raw", "state_raw"), # use temporary names
    sep = ",",                         # split by comma
    remove = FALSE,                    # keep original location column
    fill = "right",                    # handle rows with no comma
    extra = "merge"                    # handle rows with multiple commas
  ) |>
  # # step 2: trim whitespace from the raw split columns
  mutate(
    city_raw = str_trim(city_raw),
    state_raw = str_trim(state_raw)
  ) |>
  # # step 3: manually correct the few specific messy state strings
  mutate(
    state = case_when(
      state_raw == "Florida-Thomasville, Georgia" ~ "Florida",
      state_raw == "Alabama-Pensacola, Florida" ~ "Alabama",
      state_raw == "Tennessee-Virginia" ~ "Tennessee",
      TRUE ~ state_raw
    )
  ) |>
  # # step 4: extract primary city name, handling "tri-cities" specifically
  mutate(
    # split city_raw by hyphen or ampersand (needed for most dmas)
    city_parts = stringr::str_split(city_raw, pattern = "\\s*[-&]\\s*"),

    # extract the first part, but use case_when for the "tri-cities" exception
    city_primary = case_when(
      # specifically handle "tri-cities" before applying the general rule
      startsWith(city_raw, "Tri-Cities") ~ "Tri-Cities",
      # default rule: extract the first part from the split list
      TRUE ~ sapply(city_parts, function(parts) parts[1])
    ),

    # clean up the extracted primary city
    city_primary = str_remove(city_primary, ",$"),
    city_primary = str_trim(city_primary),
    state = str_trim(state)
  ) |>
  # # step 5: select final columns and remove temporary ones
  select(location, city = city_raw, state, city_primary, everything(), -city_parts) # remove city_parts here
